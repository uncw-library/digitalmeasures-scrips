# 
cd ..../example
export saxon='java -jar xslt/saxon9he.jar'

# Gather
$saxon source/EduRS.xml xslt/gather.xsl extPerIn=`pwd`/acc/Per0.xml extOrgIn=`pwd`/acc/Org0.xml > xmls/ED0.xml

# Count UnResolved Org URIs and UnResolved Person URIs
#
$saxon xmls/ED0.xml xslt/countURPs.xsl  #... 11
$saxon xmls/ED0.xml xslt/countUROs.xsl  #... 5

# Make Unique Resource Identifiers for UROs and URPs
#
bin/nuno -t EX- -Xn 11 > xmls/EX-URP-UNOs.xml
bin/nuno -t EX- -Xn 5  > xmls/EX-URO-UNOs.xml

# Make new Org list and new Person list
#
$saxon xmls/ED0.xml xslt/makeURPs.xsl unoMapFile=`pwd`/xmls/EX-URP-UNOs.xml > xmls/NewPers.xml
$saxon xmls/ED0.xml xslt/makeUROs.xsl unoMapFile=`pwd`/xmls/EX-URO-UNOs.xml > xmls/NewOrgs.xml

# Fill in the missing Org URIs and the missing Person URIs
#
$saxon xmls/ED0.xml xslt/fillURPs.xsl extPerIn=`pwd`/xmls/NewPers.xml > xmls/ED1.xml
$saxon xmls/ED1.xml xslt/fillUROs.xsl extOrgIn=`pwd`/xmls/NewOrgs.xml > xmls/ED2.xml

# Create the Education Rdf
#
$saxon xmls/ED2.xml xslt/mkEduRdf.xsl > rdf/ED.rdf

# create rdf for new Orgs and Persons
#
$saxon xmls/NewOrgs.xml xslt/mkAccOrgRdf.xsl > rdf/NewOrgs.rdf
$saxon xmls/NewPers.xml xslt/mkAccPeopleRdf.xsl > rdf/NewPers.rdf
